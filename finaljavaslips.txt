class SharedResource {
private int value;
private boolean hasValue = false; // Flag to check if value is available
// Producer method to generate a value
public synchronized void produce(int newValue) {
while (hasValue) {
try {
wait(); // Wait if consumer hasn't consumed the previous value
} catch (InterruptedException e) {
e.printStackTrace();
}
}
value = newValue;
hasValue = true;
System.out.println("Produced: " + value);

notify(); // Notify the consumer
}
// Consumer method to consume the value
public synchronized void consume() {
while (!hasValue) {
try {
wait(); // Wait if there is no value to consume
} catch (InterruptedException e) {
e.printStackTrace();
}
}
System.out.println("Consumed: " + value);
hasValue = false;
notify(); // Notify the producer
}
}
